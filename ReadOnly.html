ReadOnly

class in Core.Shared

/

Implemented in: Core.Shared.Utilities

Description

Make users unable to modify a field in inspector.

When a variable or field is public in the inspector or serialized, the user is able to modify the field at will.

For those fields that the user needs in inspector but don't want to touch add the [ReadOnly] attribute to them. Unity will show that field in the inspector grayed out. That field is visible but cannot be interacted with or modified. 

using UnityEngine;
using Core.Shared.Utilities

public class ExampleScript : MonoBehaviour
{
    //Those field can be modified in inspector
    public float modifier = 2.5f;
    [SerializeField] private int number = 2025

    //This variable is public but can not be modified in inspector
    [ReadOnly] public float hour;

    void Update()
    {
        hour += (modifier / number) * Time.deltaTime;
    }
}
